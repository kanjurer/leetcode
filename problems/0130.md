# [Problem 130: Surrounded Regions](https://leetcode.com/problems/surrounded-regions/)

## Solution

```py
# Depth first search

# Time complexity: O(n)
# Space complexity: O(n)
class Solution:
    def solve(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        
        m = len(board)
        n = len(board[0])

        visited = set()

        def dfs(i, j):
            if (i, j) in visited:
                return 
            
            visited.add((i, j))
            
            if 0 <= i < m and 0 <= j < n:
                if board[i][j] == "O":
                    board[i][j] = "E"
                    dfs(i+1, j)
                    dfs(i-1, j)
                    dfs(i, j-1)
                    dfs(i, j+1)

        for i in range(m):
            for j in [0, n-1]:
                if board[i][j] == "O":
                    dfs(i, j)
        
        for i in [0, m-1]:
            for j in range(n):
                if board[i][j] == "O":
                    dfs(i, j)
        

        for i in range(m):
            for j in range(n):
                if board[i][j] == "O":
                    board[i][j] = "X"

        for i in range(m):
            for j in range(n):
                if board[i][j] == "E":
                    board[i][j] = "O"
```
