# [Problem 706: Design HashMap](https://leetcode.com/problems/design-hashmap/)

## Solution

```py
class Node:
    key: int
    value: int
    next: Self

    def __init__(self, key=-1, value=-1, next=None):
        self.key = key
        self.value = value
        self.next = next


class MyHashMap:
    m: list[Node]
    F: int

    def __init__(self):
        self.F = 1000
        self.m = []

        for _ in range(self.F):
            self.m.append(Node())

    # Time complexity: O(n)
    # Space complexity: O(1)
    def put(self, key: int, value: int) -> None:
        i = self.hash(key)
        head = self.m[i]
        prev = None
        while head:
            if head.key == key:
                head.value = value
                return
            prev = head
            head = head.next

        prev.next = Node(key, value)

    # Time complexity: O(n)
    # Space complexity: O(1)
    def get(self, key: int) -> int:
        i = self.hash(key)
        head = self.m[i]

        while head:
            if head.key == key:
                return head.value
            head = head.next

        return -1

    # Time complexity: O(n)
    # Space complexity: O(1)
    def remove(self, key: int) -> None:
        i = self.hash(key)
        head = self.m[i]
        prev = None
        while head:
            if head.key == key:
                prev.next = head.next
                return
            prev = head
            head = head.next

    # Time complexity: O(1)
    # Space complexity: O(1)
    def hash(self, key: int) -> int:
        return key % self.F

```
